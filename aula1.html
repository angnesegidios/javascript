 Modelo de Script no javascript 

var idade = 29
var idade2 = 10
alert(idade + idade2)
// nesse caso ele irá somar as duas idades 
// dando a resposta de: 39

var idade = "29"
var idade2 = "10"
alert(idade + idade2)
// nesse caso ele irá concatenar as duas idades 
// dando a resposta de: 2910
